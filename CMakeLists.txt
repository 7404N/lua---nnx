
CMAKE_MINIMUM_REQUIRED(VERSION 2.6 FATAL_ERROR)
CMAKE_POLICY(VERSION 2.6)
FIND_PACKAGE(Torch REQUIRED)
FIND_PACKAGE(OpenMP)

IF (OPENMP_FOUND)
  MESSAGE (STATUS "OpenMP Found with compiler flag : ${OpenMP_C_FLAGS}")
  SET (CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${OpenMP_C_FLAGS}")
ENDIF (OPENMP_FOUND)

SET(src lbfgs.c)
SET(luasrc)
ADD_TORCH_PACKAGE(lbfgs "${src}" "${luasrc}" "Image Processing")
TARGET_LINK_LIBRARIES(lbfgs luaT TH)

SET(src init.c)
FILE(GLOB luasrc *.lua)
SET(luasrc ${luasrc} test/test-all.lua)
SET(luasrc ${luasrc} test/test-omp.lua)
SET(luasrc ${luasrc} test/test_lbfgs.lua)
SET(luasrc ${luasrc} test/test_cg.lua)
SET(luasrc ${luasrc} test/rosenbrock.lua)
ADD_TORCH_PACKAGE(nnx "${src}" "${luasrc}" "Image Processing")
TARGET_LINK_LIBRARIES(nnx luaT TH)
